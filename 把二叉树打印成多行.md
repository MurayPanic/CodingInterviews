# 剑指offer - 把二叉树打印成多行
题目描述： 从上到下按层打印二叉树，同一层结点从左至右输出。每一层输出一行。


## 解题思路：
- 在按层遍历二叉树的时候，新增两个变量记录下一层的节点数，以及当前层还有多少个节点需要打印；

## 解法一：

```c++
/*
struct TreeNode {
    int val;
    struct TreeNode *left;
    struct TreeNode *right;
    TreeNode(int x) :
            val(x), left(NULL), right(NULL) {
    }
};
*/
class Solution {
public:
        vector<vector<int> > Print(TreeNode* pRoot) {
            
         vector<vector<int> > result;
         if(pRoot == nullptr) {return result;}
            
         int toPrint{0};
         int nextToPrint{0};
          
          queue<TreeNode*> assist_queue;
          assist_queue.push(pRoot);
          ++toPrint;
          vector<int> currLevel;
            
          while(!assist_queue.empty()){
              auto curr = assist_queue.front();
              
              if(curr->left != nullptr){
                  assist_queue.push(curr->left);
                  ++nextToPrint;
              }
              
              if(curr->right != nullptr){
                  assist_queue.push(curr->right);
                  ++nextToPrint;
              }
              
              currLevel.push_back(curr->val);
              assist_queue.pop();
              --toPrint;
              
              if(toPrint == 0){
                  result.push_back(currLevel);
                  currLevel.clear();
                  toPrint= nextToPrint;
                  nextToPrint = 0;
              }
              
              
              
          }
            
         return result;
            
            
        }
    
};

```
